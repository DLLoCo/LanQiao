int main()
{
    int n;cin>>n;
    for(int i=1;i<=n;i++)cout<<2*max(n-i,i-1)<<endl;
}

## 优化写法:
#- 写了一堆条件判断,最后得出是4/6/8

#+ 直接三种情况归类
```py
def main():
    # 答案只有三种情况，4 6 8
    if fun4():
        print(4)
        return 
    if fun6():
        print(6)
        return 
    print(8)

def fun4():
    x1,x2,x3 = a+b,c+d,e+f
    for i in [a,b]:
        for j in [c,d]:
            for k in [e,f]:
                if i == j == k:      # 三个矩阵有共同长的边
                    return True
                if  i == j and x1+x2-i-j == k:  # 两个矩形合并后的矩形 可以和第三个矩形合并
                    return True
                if i == k and x1+x3-i-k == j:
                    return True
                if j == k and x2+x3-j-k == i:
                    return True
    return False

def fun6():
    for i in [a,b]:
        for j in [c,d]:
            for k in [e,f]:
                if i == j or j == k or i == k:
                    return True
                if i + j == k or i + k == j or j + k == i:
                    return True
    return False


n = int(input())

for _ in range(n):
    a,b,c,d,e,f = map(int,input().split())
    main()

```
