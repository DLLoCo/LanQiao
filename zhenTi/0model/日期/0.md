```py
from datetime import *
# 请在此输入您的代码
s=date(1900,1,1)
e=date(9999,12,31)
cnt=0
#! while s<=e: 超出范围,if 先加上s+=timedelta(days=1)
while s<=e:
    #! s+=timedelta(days=1)
    y,m,d=map(str,(s.year,s.month,s.day))
    y=[int(i) for i in y]
    m=[int(i) for i in m]
    d=[int(i) for i in d]
    y,m,d=sum(y),sum(m),sum(d)
    if y==m+d:
        cnt+=1
    s+=timedelta(days=1)
print(cnt)
```
